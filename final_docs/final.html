What does the standard deviation of the bootstrap averages indicate about the uncertainty in our average estimate (using question 2)? (You may filter each year and do it one by one.) [Note that normally you do not bootstrap on dependent observations but please ignore this assumption for now.]
case_avgs <- mean(chicago_data$total)

find_avg <- function(data) {
  return(mean(data, na.rm=TRUE))
}

btstrap_fxn <- function(data) {
  sim_num <- 1000
  btstrap_avgs <- c()
  #Performing bootstrapping
  for(i in 1:sim_num) {
    btstrap_sim <- sample(data,replace = TRUE)
    btstrap_avgs[i] <- find_avg(btstrap_sim)
  }
  #print(bootstrap_means)
  sd_avgs <- sd(btstrap_avgs)
  return(sd_avgs)
}

for (year in c(2020, 2021, 2022, 2023)) {
  yearTotal <- chicago_data[chicago_data$year == year, "total"]
  yearSD <- btstrap_fxn(yearTotal)
  print(paste(year, yearSD))
}
